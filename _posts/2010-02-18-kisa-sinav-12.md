int equi ( int[] A )
{
    Int64 ilkYari;
    Int64 sonYari;
    for (int iLoop = 0; iLoop &lt; A.Length; iLoop++)
    {
        ilkYari = 0;
        sonYari = 0;
        for (int yLoop = 0; yLoop &lt;= iLoop; yLoop++)
            ilkYari += A[yLoop];
        for (int zLoop = iLoop; zLoop &lt; A.Length; zLoop++)
            sonYari += A[zLoop];
        if (ilkYari == sonYari)
            return iLoop;
    }
    return -1;
}
